private fun captureEmailCodeOnSend(): () -> String {
    var capturedCode: String? = null
    doAnswer { invocation ->
        capturedCode = invocation.arguments[1] as String
        null
    }.`when`(emailService).sendCode(anyString(), anyString(), anyString())
    return { capturedCode ?: error("emailService.sendCode was not called") }
}

private fun extractVerificationId(result: MvcResult): String {
    val body = result.response.contentAsString
    val mapper = JsonMapper.builder().build()
    val tree = mapper.readTree(body)
    return tree.get("verificationId").asText()
}

1. **`forgot password success resets password and revokes tokens`**
   - Create LOCAL user with known password "OldPassword123"
   - Create a refresh token for user (via refreshTokenService.createToken or direct save)
   - `captureEmailCodeOnSend()` to capture code
   - POST `/api/v1/auth/forgot-password` with `{"email": "test@example.com"}` -> expect 202 with verificationId
   - Extract verificationId and captured code
   - POST `/api/v1/auth/reset-password` with `{"verificationId": "...", "email": "test@example.com", "code": "...", "newPassword": "NewPassword456"}` -> expect 200
   - Verify: user's passwordHash changed (passwordEncoder.matches("NewPassword456", user.passwordHash) == true)
   - Verify: all refresh tokens for user are deleted (refreshTokenRepository.findAll().none { it references user })

2. **`forgot password unknown email returns 202 with null verificationId`**
   - POST `/api/v1/auth/forgot-password` with `{"email": "unknown@example.com"}` -> expect 202
   - Response body: `{"verificationId": null}` (anti-enumeration)
   - EmailService.sendCode NOT called (verify with Mockito.verify(..., never()))

3. **`reset password with wrong code returns 401`**
   - Create LOCAL user, request forgot-password, capture code

4. **`change password success updates hash and revokes tokens`**
   - Create LOCAL user with password "OldPassword123"
   - Generate access token
   - POST `/api/v1/users/me/change-password` with Bearer token, body `{"currentPassword": "OldPassword123", "newPassword": "NewPassword456"}` -> expect 200
   - Verify: passwordEncoder.matches("NewPassword456", refreshed user's passwordHash) == true
   - Verify: refresh tokens for user are deleted

5. **`change password with wrong current password returns 401`**
   - Create LOCAL user
   - Generate access token
   - POST `/api/v1/users/me/change-password` with `{"currentPassword": "WrongPassword", "newPassword": "NewPassword456"}` -> expect 401

6. **`change password for social-only user returns 409`**
   - Create a GOOGLE-only user (no passwordHash, no LOCAL provider)

   - POST `/api/v1/users/me/change-password` with `{"currentPassword": "anything", "newPassword": "NewPassword456"}` -> expect 409
